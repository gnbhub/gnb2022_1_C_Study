#include <stdio.h> 

int main(void) {
	int i;
	int j;
	int temp; 
	int num[5] = { 0 };

	printf("값을 5개 입력해주세요:");
	for (i = 0; i < 5; i++) {
		scanf_s("%d", &num[i]);
	}

	for (i = 0; i < 5; i++) {
		for (j = 0; j < 5; j++) {
			if (num[i] < num[j]) {  
				temp = num[i];
				num[i] = num[j];
				num[j] = temp;
			}
		}
	}

	printf("정렬 결과는: ");
	for (i = 0; i < 5; i++) {
		printf("%d ", num[i]);
	}

	return 0;
}
