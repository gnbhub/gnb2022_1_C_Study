#include <stdio.h> 

int main(void) {
	int i;
	int j;
	int temp; 
	int num[5] = { 0 };

	printf("값을 5개 입력해주세요:");
	for (i = 0; i < 5; i++) {
		scanf_s("%d", &num[i]);
	}

	for (i = 0; i < 5; i++) {
		for (j = 0; j < 5; j++) {
			if (num[i] < num[j]) {  
				temp = num[i];
				num[i] = num[j];
				num[j] = temp;
			}
		}
	}

	printf("정렬 결과는: ");
	for (i = 0; i < 5; i++) {
		printf("%d ", num[i]);
	}

	return 0;
}

#include <stdio.h>           
#define d  30    

int main()
{
    int i, len = 0;
    char str[d];

    printf("문자를 입력하세요 : ");
    scanf_s("%[^\n]s", str, d - 1);

    while (str[len] != 0) len++;

    printf("로꾸거=");
    for (i = len - 1; i >= 0; i--) {
        printf("%c", str[i]);  
    }

    printf("\n");
    return 0;
}
