#include <stdio.h>
#define MAX_SIZE 5
#define _CRT_SECURE_NO_WARNINGS

// 버블 정렬
void bubble_sort(int list[]) {
	int i, j, temp;

	for (i = 4; i > 0; i--) {
		// 0 ~ (i-1)까지 반복
		for (j = 0; j < i; j++) {
			// j번째와 j+1번째의 요소가 크기 순이 아니면 교환
			if (list[j] < list[j + 1]) {
				temp = list[j];
				list[j] = list[j + 1];
				list[j + 1] = temp;
			}
		}
	}
}

int main(void) {
	int i;
	int a = 0, b = 0, c = 0, d = 0, e = 0;
	int list[5] = { a, b, c, d, e };

	scanf("%d, %d, %d, %d, %d", &a, &b, &c, &d, &e);
	
	// 버블 정렬 수행
	bubble_sort(list);

	// 정렬 결과 출력
	for (i = 0; i < 5; i++) {
		printf("%d ", list[i]);
	}
}
