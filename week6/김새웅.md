#include <stdio.h>          
#define MAX_LENGTH     50   // 최대 50자까지만 입력 받는다.

void change(char a_str[])//함수 change 만들기
{
    int i;

    for (i = 0; a_str[i] != 0; i++) {
        if (a_str[i] >= 'a' && a_str[i] <= 'z') {
            // 소문자이면 대문자로 변경
            a_str[i] = a_str[i] - 32;
        }
    }
}

int main()
{
    // 배열
    char str[100];

    printf("문자열을 입력하시오 : ");
    scanf_s("%[^\n]", str, MAX_LENGTH - 1);
    change(str);
    printf("%s\n", str);
    return 0;
}
