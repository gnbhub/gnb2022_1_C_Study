#include <stdio.h>
#define PI 3.141592 
int main() {
	float r, cir, area;;
	printf("\n원의 반지름을 입력하시오cm:"), scanf("%f", &r);
	cir = 2 * r * PI;
	area = r * r * PI;
	printf("\n원의 둘레:%.2fcm\n", cir);// 실수를 소수점 2자리까지만 출력  하는 방법!!!  printf("%.2f\n",아무거나); 
	printf("원의 넓이:%.2fcm\n", area);
	return 0;
}

#include <stdio.h>
int main() {
    int n;
    printf("소인수 분해 할 숫자를 입력하시오 : ");
    scanf_s("%d", &n);
    for (int i = 2; i <= n; i++) {
        if (n % i == 0) {
            n /= i;
            printf(" %d ", i);
            if (n > 1) {
                printf("*");
            }
        }
    }

    if (n < 1) {
        printf(" %d", n);
    }

    return 0;
}
